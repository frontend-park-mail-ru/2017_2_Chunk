 18 // Middleware
 19 app.use(morgan('dev'));                     // Формат выводимой инфы о запросах
 20 app.use(express.static('./public'));        // Отдаёт статику при совпадении имён
 21 app.use(bodyParser.json());                 // С помощью какой-то древней магии парсит тело запроса,
 22 app.use(cookieParser());                    // всё то же волшебство, но уже для кук

 24 let users = {};
 25 let ids = {};
 26 const ttl = 1000 * 60 * 60 * 24;            // 1 день
 27 
 28 app.get('/whoisit', (request, response) => {
 29 
 30 	// Вытаскиваем нужную куку
 31 	const id = request.cookies['my_cookie'];
 32 	console.log('my id: ', id);
 33 
 34 	// Устанавливаем заголовки ответа
 35 	response.set('Content-Type', 'application/json; charset=utf8');
 36 
 37 	// array[undefined] === undefined
 38 	!ids[id] ? response.status(404).end() :
 39 		response.status(200).send(JSON.stringify({username: ids[id]}));
 40 
 41 });

 51 app.post('/sign_up', (request, response) => {
 52 
 53 	const username = request.body.email;
 54 	const password = request.body.password;
 55 
 56 	console.log(username, password);
 57 
 58 	// Устанавливаем заголовок ответа
 59 	response.set('Content-Type', 'application/json; charset=utf8');
 60 
 61 	if (!username || !password) {
 62 		response.status(400).end(JSON.stringify({
 63 			errorMessage: "Логин и(или) пароль не указаны"
 64 		}));
 65 		return;
 66 	}
 67 	if (users[username]) {
 68 		return response.status(400).end(JSON.stringify({
 69 			errorMessage: ("Пользователь с именем '" + username + "' уже существует!")
 70 		}));
 71 	}
 72 
 73 	const new_id = idCreator();
 74 	users[username] = password;
 75 	ids[new_id] = username;
 76 
 77 	response.cookie('my_cookie', new_id, {          // Название и значение куки
 78 		expires: new Date(Date.now() + ttl)        // Время жизни куки
 79 	});
 80 	response.status(200).end();
 81 });

 84 app.post('/sign_in', (request, response) => {
 85 
 86 	const username = request.body.email;
 87 	const password = request.body.password;
 88 	console.log(username, password);
 89 
 90 	// Устанавливаем заголовок ответа
 91 	response.set('Content-Type', 'application/json; charset=utf8');
 92 
 93 	if (!username || !password) {
 94 		return response.status(400).send(JSON.stringify({
 95 			errorMessage: "Логин и(или) пароль не указаны"
 96 		}))
 97 	}
 98 	if (users[username] !== password) {
 99 		return response.status(400).send(JSON.stringify({
100 			errorMessage: "Неверные логин и(или) пароль"
101 		}))
102 	}
103 
104 	const new_id = idCreator();
105 	ids[new_id] = username;
106 	response.cookie('my_cookie', new_id, {
107 		expires: new Date(Date.now() + ttl)
108 	});
109 	response.status(200).end();
110 });
