1 (function () {
  2 	'use strict';
  3 
  4 	/**
  5 	 * Базовый класс формы
  6 	 * @module Form
  7 	 */
  8 	class Form extends Block {
  9 		/**
 10 		 * @param Fields []- корневой элемент блока
 11 		 * @constructor
 12 		 */
 13 		constructor(fields = []) {
 14 			const el = document.createElement('form');
 15 			super(el);
 16 
 17 			fields.forEach(function (field) {
 18 				const f = Block.Create('input', field.attrs || {}, field.classes || []);
 19 				this.append(f);
 20 			}.bind(this));
 21 		}
 22 
 23 
 24 		onSubmit(callback) {
 25 			this.el.addEventListener('submit', (event) => {
 26 				event.preventDefault();
 27 				const formdata = {};
 28 				const elements = this.el.elements;
 29 				//запись элементов формы в formdata
 30 				for (let element in elements) {
 31 					formdata[elements[element].name] = elements[element].value;
 32 				}
 33 
 34 				callback(formdata);
 35 			});
 36 		}
 37 
 38 		// errorMessage (text) {
 39 		//     this.password.value = this.confirm.value = '';
 40 		//     this.message.parentElement.hidden = false;
 41 		//     this.message.innerHTML = text;
 42 		// }
 43 
 44 		reset() {
 45 			this.el.reset();
 46 		}
 47 	}
 48 
 49 	window.Form = Form;
 50 
 51 })();
