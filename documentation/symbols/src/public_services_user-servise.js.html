1 (function () {

  7 	 * Сервис для работы с юзерами

  9 	 */

 12 			this.user = null;

 16 		/**
 17 		 * Регистрирует нового пользователя
 18 		 * @param {string} email
 19 		 * @param {string} password
 20 		 * @param {Function} callback
 21 		 * @param {Function} confirm
 22 		 */
 23 		signup(email, password, confirm, callback) {
 24 			//validation
 25 			if (email.length < 4) {
 26 				callback("Длина логина должна быть не меньше 4 символов!", null);
 27 				return;
 28 			}
 29 			if (email.length > 12) {
 30 				callback("Длина логина не должна превышать 12 символов!", null);
 31 				return;
 32 			}
 33 			if (password.length < 6) {
 34 				callback("Длина пароля должна быть не меньше 6 символов!", null);
 35 				return;
 36 			}
 37 			if (password !== confirm) {
 38 				callback("Пароли не совпадают!", null);
 39 				return;
 40 			}
 41 			if (password === email) {
 42 				callback("Логин и пароль не должны совпадать!", null);
 43 				return;
 44 			}
 45 			Http.Post('/sign_up', {email, password}, callback);
 46 		}
 47 
 48

 50 		 * Авторизация пользователя
 51 		 * @param {string} email
 52 		 * @param {string} password
 53 		 * @param {Function} callback
 54 		 */
 55 		login(email, password, callback) {
 56 			if (email.length < 4) {
 57 				callback("Длина логина должна быть не меньше 4 символов!", null);
 58 				return;
 59 			}

 61 				callback("Длина логина не должна превышать 12 символов!", null);
 62 				return;
 63 			}
 64 			if (password.length < 6) {
 65 				callback("Длина пароля должна быть не меньше 6 символов!", null);
 66 				return;
 67 			}
 68 			if (password === email) {
 69 				callback("Логин и пароль не могут совпадать!", null);
 70 				return;
 71 			}
 72 			Http.Post('/sign_in', {email, password}, callback);
 73 		}

 75 		/**

 77 		 * @return {boolean}
 78 		 */
 79 		isLoggedIn() {
 80 			// this.getData();
 81 			return !!this.user;//как эта хрень работает?
 82 		}

 84 		/**
 85 		 * Загружает данные о текущем пользователе
 86 		 * @param {Function} callback
 87 		 * @param {boolean} [force=false] - игнорировать ли кэш?
 88 		 */
 89 		getData(callback, force = false) {
 90 			if (this.isLoggedIn() && !force) {
 91 				return callback(null, this.user);
 92 			}
 93 
 94 			Http.Get('/whoisit', function (err, userdata) {
 95 				if (err) {
 96 					return callback(err, userdata);
 97 				}
 98 				this.user = userdata.username;
 99 				callback(null, userdata);
100 			}.bind(this));
101 		}
102 
103 		logout() {
104 			if (this.isLoggedIn()) {
105 				this.user = null;
106 				this.users = [];
107 				this.delCookie();
108 			}
109 		}
110 
111 		delCookie() {
112 			(function () {
113 				let xhr = new XMLHttpRequest();
114 				xhr.open('GET', '/exit', true);
115 				xhr.withCredentials = true;
116 				xhr.send();
117 			})();

120 		/**
121 		 * Загружает список всех пользователей
122 		 * @param callback
123 		 */
124 		loadUsersList(callback) {
125 			Http.Get('/users', function (err, users) {
126 				if (err) {
127 					return callback(err, users);
128 				}
129 
130 				this.users = users;
131 
132 				if (this.isLoggedIn()) {
133 					this.users = this.users.map(user => {
134 						if (user.email === this.user.email) {
135 							user.me = true;
136 						}
137 						return user;
138 					});
139 				}
140 
141 				callback(null, this.users);
142 			}.bind(this));
143 		}
144 	}
