1 (function () {
  2 	'use strict';
  3 
  4 
  5 	/**
  6 	 * Базовый класс блока
  7 	 * @module Block
  8 	 */
  9 	class Block {
 10 		/**
 11 		 * @param {HTMLElement} el - корневой элемент блока
 12 		 * @constructor
 13 		 */
 14 		constructor(el) {
 15 			this.el = el;
 16 		}
 17 
 18 		/**
 19 		 * Фабричный метод, который ползволяет удобро создавать блоки с заданными характеристиками
 20 		 * @param {string} [tagName='div'] - tagName блока
 21 		 * @param {*} [attrs={}] - объект с атрибутами блока
 22 		 * @param {string[]} [classes=[]] - список имён классов
 23 		 * @param {string|null} [text=null] - опциональный текст блока
 24 		 * @return {Block}
 25 		 * @constructor
 26 		 */
 27 		static Create(tagName = 'div', attrs = {}, classes = [], text = null) {
 28 			const el = document.createElement(tagName);
 29 			classes.forEach(function (className) {
 30 				el.classList.add(className);
 31 			});
 32 			for (let name in attrs) {
 33 				el.setAttribute(name, attrs[name]);
 34 			}
 35 			if (text) {
 36 				el.textContent = text;
 37 			}
 38 			return new Block(el);
 39 		}
 40 
 41 
 42 		/**
 43 		 * Установить новый текст для блока
 44 		 * @param {string} text
 45 		 */
 46 		setText(text) {
 47 			this.el.textContent = text;
 48 		}
 49 
 50 		/**
 51 		 * Очищает содержимое блока
 52 		 */
 53 		clear() {
 54 			this.el.innerHTML = '';
 55 		}
 56 
 57 		/**
 58 		 * Скрывает блок
 59 		 */
 60 		hide() {
 61 			this.el.setAttribute('hidden', 'hidden');
 62 		}
 63 
 64 		/**
 65 		 * Отображает блок
 66 		 */
 67 		show() {
 68 			this.el.removeAttribute('hidden');
 69 		}
 70 
 71 		/**
 72 		 * Добавляет к текущему блоку дочерний
 73 		 * @param {Block} block
 74 		 * @return {Block}
 75 		 */
 76 		append(block) {
 77 			this.el.appendChild(block.el);
 78 			return this;
 79 		}
 80 
 81 		/**
 82 		 * Позволяет подписаться на событие
 83 		 * @param {string} event
 84 		 * @param {EventListener} callback
 85 		 * @return {function(this:Block)} - функция отписки от события
 86 		 */
 87 		on(event, callback) {
 88 			this.el.addEventListener(event, callback);
 89 			return () => {
 90 				this.el.removeEventListener(event, callback);
 91 			};
 92 		}
 93 	}
 94 
 95 	window.Block = Block;
 96 
 97 })();
