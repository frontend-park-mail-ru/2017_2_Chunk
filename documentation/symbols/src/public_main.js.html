 52 	function openLogin() {
 53 		if (!sections.login.ready) {
 54 			sections.login.loginform = new Form(loginFields);
 55 			sections.signup.message = new Message();
 56 			sections.login.loginform.onSubmit(function (formdata) {
 57 				sections.signup.message.clear();
 58 				sections.signup.message.hide();
 59 				userService.login(formdata.email, formdata.password, function (err, resp) {
 60 					if (err) {
 61 						sections.signup.message.setText(err);
 62 						sections.signup.message.show();
 63 						// alert(`Some error ${err.status}: ${err.responseText}`);заменить на дивчик
 64 						return;
 65 					}
 66 
 67 					sections.login.loginform.reset();
 68 					userService.getData(function (err, resp) {
 69 						if (err) {
 70 							sections.signup.message.setText(err);
 71 							sections.signup.message.show();
 72 							return;
 73 						}
 74 						openMenu();
 75 					}, true);
 76 				});
 77 			});
 78 			sections.login
 79 				.append(sections.signup.message)
 80 				.append(Block.create('h2', {}, ['section_title'], 'Войдите'))
 81 				.append(sections.login.loginform);
 82 			sections.signup.message.hide();
 83 			sections.login.ready = true;
 84 		}
 85 		sections.hide();
 86 		backToPrevPage('signup');
 87 		if (userService.isLoggedIn()) {
 88 			return openMenu();
 89 		}
 90 		sections.login.show();
 91 	}

150 				const section = target.getAttribute('data-section');
151 				if (section === 'back') {
152 					switch (current_page) {
153 						case 'signup': {

156 							break;
157 						}
158 						case 'login': {
159 							openMenu();
160 							break;
161 						}
162 					}
163 				}
164 			});
165 			sections.back
166 				.append(sections.back.items.back);
167 			sections.back.ready = true;
168 		}
169 		sections.back.hide();
170 
171 		if (current_page === 'signup' || current_page === 'login') {
172 			sections.back.show();
173 		}

235 	function openMenu() {
236 		if (!sections.menu.ready) {
237 			sections.menu.items = {
238 				profile: Block.create('div', {'data-section': 'profile'}, ['profile', 'auth'], 'dfdfgdf'),
239 				play: Block.create('button', {'data-section': 'play'}, ['button', 'auth'], 'Играть'),
240 				signup: Block.create('button', {'data-section': 'signup'}, ['button', 'unauth'], 'Зарегистрироваться'),
241 				login: Block.create('button', {'data-section': 'login'}, ['button', 'unauth'], 'Вход'),
242 				settings: Block.create('button', {'data-section': 'settings'}, ['button', 'auth'], 'Настройки'),
243 				rules: Block.create('button', {'data-section': 'rules'}, ['button', 'unauth'], 'Правила'),
244 				scores: Block.create('button', {'data-section': 'scores'}, ['button', 'unauth'], 'Таблица лидеров'),
245 				exit: Block.create('button', {'data-section': 'exit'}, ['button', 'auth'], 'Выход'),
246 			};
247 
248 			sections.menu.on('click', function (event) {
249 				event.preventDefault();
250 				const target = event.target;
251 				const section = target.getAttribute('data-section');
252 				switch (section) {
253 					case 'login':
254 						openLogin();
255 						break;
256 					case 'signup':
257 						openSignup();
258 						break;
259 					case 'scores':
260 						openScores();
261 						break;
262 					case 'profile':
263 						openProfile();
264 						break;
265 					case 'exit':
266 						userService.logout();
267 						openMenu();
268 						break;
269 				}
270 			});
271 			sections.menu
272 				.append(sections.menu.items.profile)
273 				.append(sections.menu.items.play)
274 				.append(sections.menu.items.signup)
275 				.append(sections.menu.items.login)
276 				.append(sections.menu.items.settings)
277 				.append(sections.menu.items.rules)
278 				.append(sections.menu.items.scores)
279 				.append(sections.menu.items.exit);
280 			sections.menu.ready = true;
281 		}
282 		sections.hide();
283 
284 
285 		if (userService.isLoggedIn()) {
286 			userService.getData(function (err, resp) {
287 				sections.menu.items.profile.setText(resp);
288 			});
289 			sections.menu.items.profile.show();
290 			sections.menu.items.play.show();
291 			sections.menu.items.login.hide();
292 			sections.menu.items.signup.hide();
293 			sections.menu.items.settings.show();
294 			sections.menu.items.rules.show();
295 			sections.menu.items.scores.show();
296 			sections.menu.items.exit.show();
297 		} else {
298 			sections.menu.items.profile.hide();
299 			sections.menu.items.play.hide();
300 			sections.menu.items.login.show();
301 			sections.menu.items.signup.show();
302 			sections.menu.items.settings.hide();
303 			sections.menu.items.rules.show();
304 			sections.menu.items.scores.show();
305 			sections.menu.items.exit.hide();
306 		}
307 		sections.menu.show();
308 	}
309

311 	title.on('click', openMenu);
312 	openMenu();

315 		if (err) {
